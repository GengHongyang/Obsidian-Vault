```python
# src/main.py
"""
AI-Interview 项目入口文件
今天的目标：验证 Python 环境、类型注解和基本函数编写
"""

import os
from dotenv import load_dotenv  # 用于加载 .env 文件中的配置

# 加载 .env 文件中的环境变量
load_dotenv()


def calculate_bmi(weight_kg: float, height_m: float) -> float:
    """
    根据体重（公斤）和身高（米）计算 BMI 指数。

    Args:
        weight_kg: 体重，单位公斤。
        height_m: 身高，单位米。

    Returns:
        计算出的 BMI 值，保留一位小数。

    Raises:
        ValueError: 如果体重或身高为非正数。
    """
    if weight_kg <= 0 or height_m <= 0:
        raise ValueError("体重和身高必须为正数。")
    
    bmi = weight_kg / (height_m ** 2)
    # 保留一位小数，更符合实际使用
    return round(bmi, 1)


def get_bmi_category(bmi: float) -> str:
    """
    根据 BMI 值返回对应的体重分类。

    Args:
        bmi: BMI 指数。

    Returns:
        体重分类字符串。
    """
    if bmi < 18.5:
        return "偏瘦"
    elif 18.5 <= bmi < 24:
        return "正常"
    elif 24 <= bmi < 28:
        return "偏胖"
    else:
        return "肥胖"


if __name__ == "__main__":
    # 从 .env 文件中读取配置，演示配置管理
    # 如果 .env 中没有设置，则使用默认值
    default_weight = float(os.getenv("DEFAULT_WEIGHT", 70))
    default_height = float(os.getenv("DEFAULT_HEIGHT", 1.75))

    print(f"--- BMI 计算器 ---")
    print(f"使用默认体重: {default_weight}kg, 默认身高: {default_height}m")

    # 调用带类型注解的函数
    my_bmi = calculate_bmi(weight_kg=default_weight, height_m=default_height)
    my_category = get_bmi_category(bmi=my_bmi)

    print(f"你的 BMI 是: {my_bmi}")
    print(f"你的体重分类是: {my_category}")

    # 演示异常处理
    try:
        invalid_bmi = calculate_bmi(weight_kg=-10, height_m=1.7)
    except ValueError as e:
        print(f"捕获到预期异常: {e}")
```

# src/data_structures_demo.py
"""
数据结构实战：列表、字典与推导式
用于 AI-Interview 项目的面试题管理和配置管理
"""

import json
from pathlib import Path
from typing import Dict, List


# --- 列表与推导式 ---

def filter_python_questions(questions: List[str]) -> List[str]:
    """
    从面试题列表中过滤出包含"Python"的题目。

    Args:
        questions: 面试题列表。

    Returns:
        包含"Python"的题目列表。
    """
    # 列表推导式：遍历 questions，保留包含"Python"的元素
    return [q for q in questions if "Python" in q]


# --- 字典操作 ---

def load_config(config_dict: Dict[str, str], key: str, default: str = "") -> str:
    """
    从配置字典中安全地获取配置项。

    Args:
        config_dict: 配置字典。
        key: 配置键名。
        default: 默认值。

    Returns:
        配置值，如果键不存在则返回默认值。
    """
    # dict.get() 安全取值，不会抛出 KeyError
    return config_dict.get(key, default)


def set_default_config(config_dict: Dict[str, str], key: str, value: str) -> None:
    """
    如果配置项不存在，则设置默认值。

    Args:
        config_dict: 配置字典。
        key: 配置键名。
        value: 默认值。
    """
    # dict.setdefault() 只在键不存在时设置值
    config_dict.setdefault(key, value)


# --- 文件 I/O 与 JSON ---

def save_chat_history(history: List[Dict[str, str]], filepath: Path) -> None:
    """
    将对话历史保存为 JSON 文件。

    Args:
        history: 对话历史列表，每个元素是 {"role": str, "content": str}。
        filepath: 保存路径。

    Raises:
        IOError: 如果文件写入失败。
    """
    try:
        # 确保父目录存在
        filepath.parent.mkdir(parents=True, exist_ok=True)
        
        with open(filepath, "w", encoding="utf-8") as f:
            # ensure_ascii=False 保留中文，indent=2 格式化输出
            json.dump(history, f, ensure_ascii=False, indent=2)
        print(f"对话历史已保存到: {filepath}")
    except IOError as e:
        print(f"文件写入失败: {e}")
        raise


def load_chat_history(filepath: Path) -> List[Dict[str, str]]:
    """
    从 JSON 文件加载对话历史。

    Args:
        filepath: 文件路径。

    Returns:
        对话历史列表，如果文件不存在则返回空列表。

    Raises:
        json.JSONDecodeError: 如果 JSON 格式错误。
    """
    if not filepath.exists():
        print(f"文件不存在，返回空历史: {filepath}")
        return []
    
    try:
        with open(filepath, "r", encoding="utf-8") as f:
            history = json.load(f)
        print(f"已从 {filepath} 加载 {len(history)} 条对话记录")
        return history
    except json.JSONDecodeError as e:
        print(f"JSON 解析失败: {e}")
        # 返回空列表而不是崩溃，让调用方自行处理
        return []


if __name__ == "__main__":
    # --- 列表推导式演示 ---
    print("=" * 50)
    print("1. 列表推导式演示")
    print("=" * 50)
    
    questions: List[str] = [
        "Python 中 GIL 是什么？",
        "C++ 虚函数的工作原理？",
        "Python 装饰器怎么用？",
        "什么是 RESTful API？",
        "Python 异步编程 vs 多线程",
    ]
    
    python_questions = filter_python_questions(questions)
    print(f"原始题目数: {len(questions)}")
    print(f"Python 相关题目数: {len(python_questions)}")
    for q in python_questions:
        print(f"  - {q}")
    
    # --- 字典操作演示 ---
    print("\n" + "=" * 50)
    print("2. 字典操作演示")
    print("=" * 50)
    
    config: Dict[str, str] = {
        "model": "gpt-4",
        "temperature": "0.7",
        "max_tokens": "2048",
    }
    
    # 安全取值
    model = load_config(config, "model", "gpt-3.5-turbo")
    api_key = load_config(config, "api_key", "未设置")
    print(f"当前模型: {model}")
    print(f"API Key: {api_key}")
    
    # 设置默认值
    set_default_config(config, "api_key", "sk-default-key")
    print(f"设置默认 API Key 后: {config['api_key']}")
    
    # --- 文件 I/O 演示 ---
    print("\n" + "=" * 50)
    print("3. 文件 I/O 与 JSON 演示")
    print("=" * 50)
    
    # 模拟对话历史
    chat_history: List[Dict[str, str]] = [
        {"role": "user", "content": "Python 中 GIL 是什么？"},
        {"role": "assistant", "content": "GIL 是全局解释器锁..."},
        {"role": "user", "content": "怎么绕过 GIL？"},
        {"role": "assistant", "content": "可以使用多进程或异步编程..."},
    ]
    
    # 保存到 data/ 目录

    save_chat_history(chat_history, history_file)

    print(f"重新加载后共 {len(loaded_history)} 条记录")
    for msg in loaded_history:
        print(f"  [{msg['role']}]: {msg['content'][:30]}...")

## 知识点笔记

### 1. [[Python 虚拟环境]]

**定义：** Python 的虚拟环境 (`venv`) 是一个独立的 Python 运行环境，用于隔离不同项目的依赖包，避免全局包冲突。

**为什么需要：**
- 项目 A 依赖 `httpx==0.27`，项目 B 依赖 `httpx==0.28`，全局安装会导致冲突。
- 部署时可以通过 `requirements.txt` 精确复现环境。
- 避免污染系统 Python。

**核心命令：**
```bash
# 创建虚拟环境
python -m venv venv

# 激活（Linux/Mac）
source venv/bin/activate

# 激活（Windows）
venv\Scripts\activate

# 退出
deactivate

# 导出依赖
pip freeze > requirements.txt

# 安装依赖
pip install -r requirements.txt
```

**与 C++ 对比：**
- C++ 的 `CMakeLists.txt` 管理构建配置和链接库，`vcpkg`/`conan` 管理第三方库。
- Python 的 `venv` + `requirements.txt` 只管理运行时依赖，不涉及编译。
- 两者目标一致：**可复现的构建/运行环境**。

**常见坑：**
- 忘记激活虚拟环境，包装到全局 Python。
- 激活后换了终端窗口，需要重新激活。
- 把 `venv/` 目录提交到 Git（应在 `.gitignore` 中添加）。

### 2. [[Python 类型注解]]

**定义：** Python 3.6+ 引入的可选类型注解语法，允许为变量、函数参数和返回值声明类型。

**基本语法：**
```python
# 函数注解
def greet(name: str) -> str:
    return f"Hello, {name}"

# 变量注解
age: int = 25
names: list[str] = ["Alice", "Bob"]

# 复杂类型（需从 typing 导入）
from typing import Dict, List, Optional, Union

config: Dict[str, str] = {"key": "value"}
items: List[int] = [1, 2, 3]
maybe_name: Optional[str] = None  # 等价于 Union[str, None]
```

**为什么 Agent 工程必须用：**
- 代码自文档化，新人接手能快速理解数据结构。
- IDE 提供智能补全和类型检查，减少运行时 Bug。
- Pydantic 模型（Week 3 会学）依赖类型注解做数据校验。

**常见坑：**
- 认为类型注解是多余的，只在参数上加，忘记返回值。
- 使用 `list` 而不是 `List`（Python 3.9+ 支持 `list[str]` 直接写法）。
- 过度使用 `Any` 导致注解失去意义。

### 3. [[Python 列表]]

**定义：** Python 内置的可变有序序列，用 `[]` 表示，可以存储任意类型的元素。

**核心操作：**
```python
# 创建
fruits: list[str] = ["apple", "banana", "cherry"]

# 增
fruits.append("date")        # 末尾添加一个元素
fruits.extend(["elderberry", "fig"])  # 合并另一个列表
fruits.insert(0, "apricot")  # 指定位置插入

# 删
fruits.remove("banana")      # 删除第一个匹配项
popped = fruits.pop()        # 删除并返回末尾元素
del fruits[0]                # 删除指定索引

# 改
fruits[0] = "blueberry"

# 查
if "apple" in fruits:        # 成员检查
    print("found")
index = fruits.index("cherry")  # 查找索引

# 排序
fruits.sort()                # 原地排序
sorted_fruits = sorted(fruits)  # 返回新列表
```

**列表推导式：**
```python
# 基本形式：[expression for item in iterable if condition]
squares = [x**2 for x in range(10) if x % 2 == 0]
# 结果：[0, 4, 16, 36, 64]

# 嵌套推导式
matrix = [[1, 2, 3], [4, 5, 6], [7, 8, 9]]
flattened = [num for row in matrix for num in row]
# 结果：[1, 2, 3, 4, 5, 6, 7, 8, 9]
```

**与 C++ `std::vector` 对比：**
- Python 列表是动态数组，类似 `std::vector`。
- Python 列表可以存储不同类型（不推荐），C++ vector 只能存同类型。
- Python 列表推导式等价于 C++ 的 `std::transform` + `std::copy_if`。

### 4. [[Python 字典]]

**定义：** Python 内置的键值对映射类型，用 `{}` 表示，键必须是不可变类型（字符串、数字、元组）。

**核心操作：**
```python
# 创建
config: dict[str, str] = {
    "model": "gpt-4",
    "temperature": "0.7",
}

# 增/改
config["max_tokens"] = "2048"  # 键不存在则新增，存在则覆盖
config.update({"api_key": "sk-xxx", "model": "gpt-4-turbo"})

# 删
del config["api_key"]
popped = config.pop("temperature")  # 删除并返回值

# 查（安全方式）
model = config.get("model", "gpt-3.5-turbo")  # 键不存在返回默认值
api_key = config.get("api_key")  # 键不存在返回 None

# 设置默认值
config.setdefault("api_key", "sk-default")  # 键不存在才设置

# 遍历
for key, value in config.items():
    print(f"{key}: {value}")

for key in config.keys():     # 等价于直接 for key in config
    print(key)

for value in config.values():
    print(value)
```

**与 C++ `std::map`/`std::unordered_map` 对比：**
- Python 字典是哈希表实现，类似 `std::unordered_map`。
- Python 字典的键必须是可哈希的（不可变类型）。
- `dict.get()` 比 `map[key]` 更安全，不会抛出异常。

### 5. [[JSON 序列化]]

**定义：** JSON（JavaScript Object Notation）是一种轻量级数据交换格式。Python 的 `json` 模块提供 `dump`/`dumps`（序列化）和 `load`/`loads`（反序列化）。

**核心用法：**
```python
import json
from pathlib import Path

# Python 对象 -> JSON 字符串
data = {"name": "Alice", "age": 25, "skills": ["Python", "FastAPI"]}
json_str = json.dumps(data, ensure_ascii=False, indent=2)
print(json_str)

# JSON 字符串 -> Python 对象
parsed = json.loads(json_str)
print(parsed["name"])  # Alice

# 写入文件
with open("data.json", "w", encoding="utf-8") as f:
    json.dump(data, f, ensure_ascii=False, indent=2)

# 读取文件
with open("data.json", "r", encoding="utf-8") as f:
    loaded = json.load(f)
```

**常见可序列化类型：**
| Python 类型 | JSON 类型 |
|------------|-----------|
| `dict` | `object` |
| `list`/`tuple` | `array` |
| `str` | `string` |
| `int`/`float` | `number` |
| `bool` | `true`/`false` |
| `None` | `null` |

**不可序列化类型（需要自定义处理）：**
- `datetime`、`Decimal`、`Path`、自定义类实例
- 解决方案：继承 `json.JSONEncoder` 或使用 `default` 参数

**在 Agent 工程中的应用：**
- 对话历史持久化（`chat_history.json`）
- 配置文件加载（`config.json`）
- API 请求/响应数据交换
- RAG 知识库文档存储
