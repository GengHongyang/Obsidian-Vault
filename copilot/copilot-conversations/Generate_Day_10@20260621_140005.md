### 1. 将 LLM Client 接入 FastAPI 后端

- **它是什么**：把 Week 1 的 LLM 调用能力今天正式绑定到 Week 2 的 FastAPI 骨架中，实现「用户发消息 → 保存到数据库 → 调用 LLM API → AI 回复 → 保存到数据库 → 返回给用户」的完整闭环。

- **它是什么**：通过 Pydantic Settings 统一管理模型名称、API Key、Base URL 等配置，在 `LLMClient` 中读取这些配置并增加超时、重试、错误分类（网络错误 vs API 错误 vs 模型配额不足）。

# app/llm_client.py
"""LLM API 调用封装，支持同步与异步调用"""

import httpx
import logging
from typing import Optional
from app.config import settings

logger = logging.getLogger(__name__)


class LLMClient:
    """封装 OpenAI-compatible API 的调用逻辑

    使用方式：
        client = LLMClient()
        reply = await client.chat([
            {"role": "system", "content": "你是一个面试助手"},
            {"role": "user", "content": "请出题"},
        ])
    """

    def __init__(self, timeout: int = 60):
        self.api_key = settings.llm_api_key
        self.base_url = settings.llm_base_url
        self.model = settings.llm_model
        self.timeout = timeout
        # 构建完整的 chat endpoint URL
        self.chat_url = f"{self.base_url.rstrip('/')}/v1/chat/completions"

    async def chat(
        self,
        messages: list[dict],
        temperature: float = 0.7,
        max_tokens: int = 2048,
    ) -> str:
        """发送聊天消息并返回回复文本

        Args:
            messages: messages 列表，格式为 [{"role": "user", "content": "你好"}]
            temperature: 采样温度，0-2 之间
            max_tokens: 最大输出 token 数

        Returns:
            AI 回复的文本内容

        Raises:
            httpx.TimeoutException: 请求超时
            httpx.HTTPStatusError: API 返回非 2xx 状态码
        """
        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json",
        }
        payload = {
            "model": self.model,
            "messages": messages,
            "temperature": temperature,
            "max_tokens": max_tokens,
        }

        async with httpx.AsyncClient(timeout=self.timeout) as client:
            response = await client.post(
                self.chat_url,
                headers=headers,
                json=payload,
            )
            # 自动抛出 HTTP 错误
            response.raise_for_status()
            result = response.json()
            # 提取回复文本
            reply = result["choices"][0]["message"]["content"]
            return reply
```

# app/routers/chat.py（更新版本）
"""聊天相关路由：已集成真实 LLM 调用"""

from fastapi import APIRouter, Depends, HTTPException
import httpx
import logging
from uuid import uuid4, UUID
from app.schemas.chat import ChatRequest, ChatResponse
from app.models.conversation import Conversation
from app.models.message import Message
from app.dependencies import get_settings, get_llm_client
from app.llm_client import LLMClient
from app.config import Settings

router = APIRouter(prefix="/chat", tags=["chat"])
logger = logging.getLogger(__name__)

# 系统提示词，告诉 LLM 扮演面试助手
SYSTEM_PROMPT = {
    "role": "system",
    "content": "你是一个 AI Agent 面试助手。你擅长帮助求职者准备 AI 应用开发面试。"
               "请用中文回答，回答要简洁、有深度、有代码示例。"
               "如果用户问的是概念，给出定义加例子；如果是代码问题，给出可运行的代码。"
}


@router.post("", response_model=ChatResponse)
async def create_chat(
    request: ChatRequest,
    settings: Settings = Depends(get_settings),
    llm_client: LLMClient = Depends(get_llm_client),
) -> ChatResponse:
    """处理用户消息：调用真实 LLM 并持久化对话"""
    # 1. 获取或创建 Conversation（与 Day 9 相同）
    if request.conversation_id:
        try:
            conv_id = UUID(request.conversation_id)
            conversation = await Conversation.get_or_none(id=conv_id)
            if conversation is None:
                raise HTTPException(status_code=404, detail="对话未找到")
        except ValueError:
            raise HTTPException(status_code=400, detail="无效的对话 ID 格式")
    else:
        title = request.message[:20] + "..." if len(request.message) > 20 else request.message
        conversation = await Conversation.create(title=title)

    # 2. 保存用户消息到数据库
    await Message.create(
        conversation=conversation,
        role="user",
        content=request.message,
    )

    # 3. 从数据库读取当前对话的完整历史（按时间排序）
    db_messages = await Message.filter(conversation=conversation).order_by("created_at")

    # 4. 构建 LLM 的 messages 列表：system + 历史消息
    messages = [SYSTEM_PROMPT]
    for msg in db_messages:
        messages.append({
            "role": msg.role,        # "user" 或 "assistant"
            "content": msg.content,
        })

    # 5. 调用 LLM API（带异常处理）
    try:
        reply = await llm_client.chat(messages=messages)
    except httpx.TimeoutException:
        logger.error(f"LLM API 超时 (timeout={llm_client.timeout}s)")
        raise HTTPException(status_code=504, detail="LLM 请求超时，请稍后重试")
    except httpx.HTTPStatusError as e:
        status_code = e.response.status_code
        error_body = e.response.text[:200]
        logger.error(f"LLM API 返回错误 {status_code}: {error_body}")
        if status_code == 401:
            raise HTTPException(status_code=502, detail="LLM API 认证失败，请检查 API Key")
        elif status_code == 429:
            raise HTTPException(status_code=502, detail="LLM API 请求频率过高，请稍后重试")
        else:
            raise HTTPException(status_code=502, detail=f"LLM API 返回错误 ({status_code})")
    except Exception as e:
        logger.exception(f"调用 LLM 时发生未知错误: {e}")
        raise HTTPException(status_code=500, detail="AI 服务内部错误，请联系管理员")

    # 6. 保存 AI 回复到数据库
    await Message.create(
        conversation=conversation,
        role="assistant",
        content=reply,
    )

    # 7. 返回响应
    return ChatResponse(
        reply=reply,
        conversation_id=str(conversation.id),
        model=settings.llm_model,
    )
```

"""公共依赖项，供路由注入使用"""

from fastapi import Request
from app.config import settings
from app.llm_client import LLMClient


async def get_settings(request: Request) -> "Settings":
    """注入 Settings 单例"""
    return settings


async def get_llm_client(request: Request) -> LLMClient:
    """每次请求创建一个 LLMClient 实例

    为什么不用单例？
    因为 LLMClient 内部包含 httpx.AsyncClient，每次请求创建可以避免连接池堆积。
    如果后续需要连接池复用，可以改为从 request.app.state 读取。
    """
    return LLMClient()
