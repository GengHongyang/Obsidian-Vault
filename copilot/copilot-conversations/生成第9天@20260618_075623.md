### 1. JWT 认证原理与 FastAPI 集成（[[JWT 认证]]）

- **它是什么**：JWT（JSON Web Token）是一种无状态认证协议。用户登录后服务端签发一个 token，客户端后续请求带上 token，服务端验证 token 即可确认身份，无需查数据库。
- **它解决 Agent 工程里的什么痛点**：你的 Chat API 现在任何人都能调用。Agent 后端需要区分用户、限制未授权访问、记录每个用户的对话历史。JWT 是实现这些的基础。
- **今天为什么必须学**：Week 2 核心产出物是"Agent 后端服务骨架"，认证是骨架的脊柱。没有认证，后续 RAG、面试系统、用户管理全都无法与具体用户绑定。
- **新手最容易踩什么坑**：`alg=none` 攻击（服务端未校验算法）、token 未设置过期时间、将 JWT 存入 localStorage 而不是 httpOnly cookie、混淆 `access_token` 和 `refresh_token` 的使用场景。

### 2. 密码哈希与用户注册（[[密码哈希与 bcrypt]]）

- **它是什么**：明文存储密码是严重安全违规。使用 `bcrypt` 或 `argon2` 这类哈希算法，即使数据库泄露，攻击者也拿不到原始密码。
- **它解决 Agent 工程里的什么痛点**：AI-Interview 项目需要用户注册/登录。如果你在项目中说"密码用 md5 存储"，面试官会直接挂电话。
- **今天为什么必须学**：认证不只是发 token，还包括安全的密码存储。今天你同时完成注册和登录两个端点，形成完整闭环。
- **新手最容易踩什么坑**：自己写哈希函数而不是用经过审计的标准库；不对密码做长度/复杂度校验；忘记在注册时做邮箱/用户名唯一性检查。

```python
# app/core/security.py
"""JWT token 签发/验证 和 密码哈希"""
from datetime import datetime, timedelta, timezone
from typing import Any

from jose import JWTError, jwt
from passlib.context import CryptContext

from app.core.config import settings

# passlib 上下文，默认使用 bcrypt
pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")

# JWT 配置
ALGORITHM = "HS256"
ACCESS_TOKEN_EXPIRE_MINUTES = 30  # token 30 分钟过期


def hash_password(password: str) -> str:
    """对明文密码进行 bcrypt 哈希"""
    return pwd_context.hash(password)


def verify_password(plain_password: str, hashed_password: str) -> bool:
    """验证明文密码是否匹配哈希值"""
    return pwd_context.verify(plain_password, hashed_password)


def create_access_token(
    data: dict[str, Any],
    expires_delta: timedelta | None = None,
) -> str:
    """
    生成 JWT access token。

    参数:
        data: 要编码到 token 中的数据，必须包含 "sub"（用户标识）
        expires_delta: 过期时间，默认 30 分钟
    返回:
        str: 编码后的 JWT token
    """
    to_encode = data.copy()
    expire = datetime.now(timezone.utc) + (
        expires_delta or timedelta(minutes=ACCESS_TOKEN_EXPIRE_MINUTES)
    )
    to_encode.update({"exp": expire})
    # jwt.encode 使用密钥和算法对 payload 签名
    encoded_jwt = jwt.encode(to_encode, settings.secret_key, algorithm=ALGORITHM)
    return encoded_jwt


def verify_token(token: str) -> dict[str, Any]:
    """
    验证 JWT token 并返回 payload。

    参数:
        token: 待验证的 JWT 字符串
    返回:
        dict: token 中的 payload 数据
    抛出:
        JWTError: 如果 token 无效或已过期
    """
    try:
        payload = jwt.decode(
            token, settings.secret_key, algorithms=[ALGORITHM]
        )
        return payload
    except JWTError:
        # 统一抛出 JWTError，上层捕获后返回 401
        raise
```

```python
# app/routers/auth.py
"""认证路由：注册、登录、当前用户"""
from fastapi import APIRouter, Depends, HTTPException, status
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
from jose import JWTError

from app.schemas.user import UserCreate, UserResponse, LoginRequest, Token
from app.core.security import (
    hash_password,
    verify_password,
    create_access_token,
    verify_token,
)

router = APIRouter(prefix="/api/v1/auth", tags=["Authentication"])
security_scheme = HTTPBearer()  # 从 Authorization header 提取 Bearer token

# 临时内存存储，明天替换为数据库
users_db: dict[str, dict] = {}  # key: username
user_id_counter: int = 0


async def get_current_user(
    credentials: HTTPAuthorizationCredentials = Depends(security_scheme),
) -> dict:
    """
    依赖函数：从 token 中提取当前用户。

    如果 token 无效或过期，抛出 401 异常。
    此函数可直接注入到需要认证的路由中。
    """
    token = credentials.credentials
    try:
        payload = verify_token(token)
        username: str | None = payload.get("sub")
        if username is None:
            raise HTTPException(
                status_code=status.HTTP_401_UNAUTHORIZED,
                detail="token 中缺少用户标识",
            )
    except JWTError:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="token 无效或已过期",
        )

    user = users_db.get(username)
    if user is None:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="用户不存在",
        )
    return user


@router.post("/register", response_model=UserResponse, status_code=status.HTTP_201_CREATED)
async def register(user_in: UserCreate):
    """注册新用户"""
    # 检查用户名是否已存在
    if user_in.username in users_db:
        raise HTTPException(
            status_code=status.HTTP_409_CONFLICT,
            detail="用户名已被注册",
        )

    # 检查邮箱是否已存在
    for u in users_db.values():
        if u["email"] == user_in.email:
            raise HTTPException(
                status_code=status.HTTP_409_CONFLICT,
                detail="邮箱已被注册",
            )

    global user_id_counter
    user_id_counter += 1

    hashed_pw = hash_password(user_in.password)
    user_record = {
        "id": user_id_counter,
        "username": user_in.username,
        "email": user_in.email,
        "hashed_password": hashed_pw,
        "is_active": True,
    }
    users_db[user_in.username] = user_record

    return UserResponse(
        id=user_record["id"],
        username=user_record["username"],
        email=user_record["email"],
        is_active=user_record["is_active"],
    )


@router.post("/login", response_model=Token)
async def login(login_in: LoginRequest):
    """用户登录，返回 access token"""
    user = users_db.get(login_in.username)
    if not user or not verify_password(login_in.password, user["hashed_password"]):
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="用户名或密码错误",
        )

    # 创建 token，将用户名作为 subject
    access_token = create_access_token(data={"sub": user["username"]})
    return Token(access_token=access_token)
```

```python
from fastapi import APIRouter, Depends, HTTPException
from app.schemas.chat import ChatRequest, ChatResponse

async def get_llm_client() -> LLMClient:
    from app.main import app
    return app.state.llm_client


@router.post("", response_model=ChatResponse)
async def chat(
    request: ChatRequest,
    current_user: dict = Depends(get_current_user),  # 新增认证依赖
    llm: LLMClient = Depends(get_llm_client),
):
    """发送消息给 LLM（需要登录）"""
    messages_dict = [msg.model_dump() for msg in request.messages]
    try:
        result = await llm.chat(messages_dict, temperature=request.temperature)
        reply = result["choices"][0]["message"]["content"]
        usage = result.get("usage")
        # 可以记录 current_user 信息到日志
        return ChatResponse(reply=reply, usage=usage)
    except LLMAPIError as e:
        raise HTTPException(status_code=502, detail=str(e))
```
